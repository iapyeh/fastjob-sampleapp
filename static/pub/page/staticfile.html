Static Files

Serving .html, .jpg, .js, .css files.

fastjob.Router.File() is for serving static files.

Example from unittest/route.go

import(
    fastjob "github.com/iapyeh/fastjob"
)

Router := fastjob.Router
cwd, _ := os.Getwd()

// Expose folder "../fastjob/static/file" as url "/fastjob"
// For public accessible
Router.File("/fastjob", filepath.Join(cwd, "..", "fastjob", "static", "file"), fastjob.PublicMode)

// Expose folder "./pri" as url "/unittest/pri"
// For authenticated users only
Router.File("/unittest/pri", filepath.Join(cwd, "static", "pri"), fastjob.ProtectMode)

// ... other routes

// Put the "/" at bottom (lowest priority)
Router.File("/", filepath.Join(cwd, "static", "pub"), fastjob.PublicMode)
fastjob.Router.File(UrlPath, FolderPath, ACL)
